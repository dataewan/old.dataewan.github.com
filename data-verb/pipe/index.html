Pipes in pandas

pipe

import statsmodels.formula.api as sm

bb = pd.read_csv('data/baseball.csv', index_col='id')

(bb.query('h > 0')
   .assign(ln_h = lambda df: np.log(df.h))
   .pipe((sm.poisson, 'data'), 'hr ~ ln_h + year + g + C(lg)')
   .fit()
   .summary()
)
